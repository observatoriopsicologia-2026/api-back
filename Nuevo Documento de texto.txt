CREATE DATABASE sistema_academico;
USE sistema_academico;

CREATE TABLE users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL,
    email VARCHAR(150) UNIQUE NOT NULL,
    password VARCHAR(255) NOT NULL,
    rol ENUM('admin', 'editor') DEFAULT 'admin',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE publications (
    id INT AUTO_INCREMENT PRIMARY KEY,
    titulo VARCHAR(255) NOT NULL,
    descripcion TEXT,
    tipo ENUM('Articulo', 'Libro', 'Capitulo', 'Tesis', 'Ponencia') NOT NULL,
    archivo_url VARCHAR(500) NOT NULL,
    autor VARCHAR(255) NOT NULL,
    fecha_publicacion DATE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE researchers (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(150) NOT NULL,
    titulo VARCHAR(50), -- Dra., Dr., etc
    institucion VARCHAR(255),
    pais VARCHAR(100),
    aptitudes TEXT,
    email VARCHAR(150),
    perfil_url VARCHAR(500),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE events (
    id INT AUTO_INCREMENT PRIMARY KEY,
    titulo VARCHAR(255) NOT NULL,
    descripcion TEXT,
    fecha_inicio DATETIME NOT NULL,
    fecha_fin DATETIME,
    ubicacion VARCHAR(255),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE news (
    id INT AUTO_INCREMENT PRIMARY KEY,
    titulo VARCHAR(255) NOT NULL,
    contenido TEXT NOT NULL,
    tipo ENUM('Noticia', 'Novedad') NOT NULL,
    imagen_url VARCHAR(500),
    fecha_publicacion DATE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Publicaciones
CREATE INDEX idx_publications_titulo ON publications(titulo);
CREATE INDEX idx_publications_autor ON publications(autor);
CREATE INDEX idx_publications_tipo ON publications(tipo);

-- Investigadores
CREATE INDEX idx_researchers_nombre ON researchers(nombre);
CREATE INDEX idx_researchers_pais ON researchers(pais);

-- Eventos
CREATE INDEX idx_events_fecha_inicio ON events(fecha_inicio);

-- Noticias
CREATE INDEX idx_news_tipo ON news(tipo);
